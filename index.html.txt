<!DOCTYPE html>
<html lang="zh-TW">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>2026 大港時程表</title>
    <meta name="apple-mobile-web-app-capable" content="yes">
    <meta name="apple-mobile-web-app-status-bar-style" content="black-translucent">
    <meta name="apple-mobile-web-app-title" content="大港時程表">
    
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script src="https://cdn.tailwindcss.com"></script>
    <script src="https://unpkg.com/lucide@latest"></script>
    
    <style>
        .no-scrollbar::-webkit-scrollbar { display: none; }
        .no-scrollbar { -ms-overflow-style: none; scrollbar-width: none; }
        body { -webkit-tap-highlight-color: transparent; }
    </style>
</head>
<body>
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;
        const { 
            Settings, Plus, Trash2, Calendar, Clock, MapPin, 
            CheckCircle2, AlertTriangle, LayoutGrid, List, X, Save 
        } = lucide;

        // --- Constants ---
        const TIME_STEP = 10; 
        const PRESET_COLORS = ['#EF4444', '#F97316', '#F59E0B', '#10B981', '#06B6D4', '#3B82F6', '#8B5CF6', '#EC4899', '#64748B', '#000000'];

        const timeToMinutes = (time) => {
            const [h, m] = time.split(':').map(Number);
            return h * 60 + m;
        };

        const minutesToTime = (totalMinutes) => {
            const h = Math.floor(totalMinutes / 60);
            const m = totalMinutes % 60;
            return `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}`;
        };

        // --- Main App Component ---
        function App() {
            const [stages, setStages] = useState([]);
            const [acts, setActs] = useState([]);
            const [selectedIds, setSelectedIds] = useState([]);
            const [activeTab, setActiveTab] = useState('explore');
            const [isAdminOpen, setIsAdminOpen] = useState(false);
            const [currentDay, setCurrentDay] = useState('');
            const [activeStageId, setActiveStageId] = useState('');

            useEffect(() => {
                const savedStages = localStorage.getItem('fest_v3_stages');
                const savedActs = localStorage.getItem('fest_v3_acts');
                const savedPicks = localStorage.getItem('fest_v3_picks');
                if (savedStages) setStages(JSON.parse(savedStages));
                if (savedActs) setActs(JSON.parse(savedActs));
                if (savedPicks) setSelectedIds(JSON.parse(savedPicks));
                lucide.createIcons();
            }, []);

            useEffect(() => {
                localStorage.setItem('fest_v3_stages', JSON.stringify(stages));
                localStorage.setItem('fest_v3_acts', JSON.stringify(acts));
                localStorage.setItem('fest_v3_picks', JSON.stringify(selectedIds));
                lucide.createIcons();
            }, [stages, acts, selectedIds]);

            const availableDays = useMemo(() => Array.from(new Set(acts.map(a => a.day))).sort(), [acts]);
            
            useEffect(() => {
                if (!currentDay && availableDays.length > 0) setCurrentDay(availableDays[0]);
                if (!activeStageId && stages.length > 0) setActiveStageId(stages[0].id);
            }, [availableDays, stages]);

            const dayRange = useMemo(() => {
                const dayActs = acts.filter(a => a.day === currentDay);
                if (dayActs.length === 0) return { start: 660, end: 1320 };
                const starts = dayActs.map(a => timeToMinutes(a.startTime));
                const ends = dayActs.map(a => timeToMinutes(a.endTime));
                return {
                    start: Math.floor(Math.min(...starts) / 60) * 60,
                    end: Math.ceil(Math.max(...ends) / 60) * 60
                };
            }, [acts, currentDay]);

            const togglePick = (id) => setSelectedIds(prev => prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id]);

            return (
                <div className="min-h-screen bg-slate-50 text-slate-900 pb-24 antialiased font-sans">
                    <header className="sticky top-0 z-50 bg-white border-b p-4 flex justify-between items-center shadow-sm">
                        <div>
                            <h1 className="font-black text-xl tracking-tight">2026 大港時程表</h1>
                            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest flex items-center gap-1">
                                <span className="w-1.5 h-1.5 bg-orange-500 rounded-full animate-pulse"></span> Personal Edition
                            </p>
                        </div>
                        <button onClick={() => setIsAdminOpen(true)} className="p-2.5 bg-slate-100 rounded-xl"><Settings size={20} /></button>
                    </header>

                    <nav className="flex bg-white border-b sticky top-[69px] z-40">
                        <TabBtn active={activeTab === 'explore'} onClick={() => setActiveTab('explore')} icon="layout-grid" label="舞台" />
                        <TabBtn active={activeTab === 'matrix'} onClick={() => setActiveTab('matrix')} icon="calendar" label="總表" />
                        <TabBtn active={activeTab === 'picks'} onClick={() => setActiveTab('picks')} icon="list" label="時程表" />
                    </nav>

                    {availableDays.length > 0 && (
                        <div className="px-4 py-3 flex gap-2 overflow-x-auto no-scrollbar bg-white/50 backdrop-blur-sm border-b">
                            {availableDays.map(day => (
                                <button key={day} onClick={() => setCurrentDay(day)} className={`px-4 py-1.5 rounded-full text-xs font-black transition-all border-2 ${currentDay === day ? 'bg-black border-black text-white' : 'bg-white border-slate-200 text-slate-400'}`}>{day}</button>
                            ))}
                        </div>
                    )}

                    <main className="p-4 max-w-md mx-auto">
                        {activeTab === 'explore' && (
                            <div className="space-y-4">
                                <div className="flex overflow-x-auto gap-2 no-scrollbar pb-2">
                                    {stages.map(s => (
                                        <button key={s.id} onClick={() => setActiveStageId(s.id)} className={`px-4 py-2 rounded-xl text-xs font-black whitespace-nowrap border-2 transition-all flex items-center gap-2 ${activeStageId === s.id ? 'bg-white border-slate-900' : 'bg-white border-slate-100 text-slate-400'}`}>
                                            <div className="w-2 h-2 rounded-full" style={{ backgroundColor: s.color }} />{s.name}
                                        </button>
                                    ))}
                                </div>
                                <div className="space-y-3">
                                    {acts.filter(a => a.day === currentDay && a.stageId === activeStageId).sort((a,b) => timeToMinutes(a.startTime) - timeToMinutes(b.startTime)).map(act => (
                                        <div key={act.id} onClick={() => togglePick(act.id)} className={`p-5 rounded-[24px] border-2 transition-all flex items-center justify-between ${selectedIds.includes(act.id) ? 'border-slate-900 bg-white shadow-xl' : 'border-transparent bg-white shadow-sm'}`}>
                                            <div className="space-y-1">
                                                <div className="text-[10px] font-black font-mono px-2 py-0.5 rounded-md inline-block" style={{ backgroundColor: `${stages.find(s=>s.id===act.stageId)?.color}20`, color: stages.find(s=>s.id===act.stageId)?.color }}>{act.startTime} — {act.endTime}</div>
                                                <h3 className="font-bold text-lg">{act.name}</h3>
                                            </div>
                                            <div className={`w-8 h-8 rounded-full border-2 flex items-center justify-center ${selectedIds.includes(act.id) ? 'bg-black border-black text-white' : 'border-slate-100'}`}>
                                                {selectedIds.includes(act.id) && <CheckCircle2 size={16} />}
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        )}

                        {activeTab === 'matrix' && (
                            <div className="overflow-x-auto border-2 border-slate-900 rounded-[32px] bg-white shadow-2xl relative overflow-hidden">
                                <div className="grid" style={{ gridTemplateColumns: `60px repeat(${stages.length}, 100px)`, gridTemplateRows: `50px repeat(${(dayRange.end - dayRange.start) / TIME_STEP}, 25px)` }}>
                                    <div className="sticky top-0 left-0 bg-slate-900 z-30 flex items-center justify-center text-[10px] text-white font-black italic">TIME</div>
                                    {stages.map(s => (
                                        <div key={s.id} className="sticky top-0 bg-slate-900 text-white text-[10px] font-black flex flex-col items-center justify-center z-20 border-l border-slate-800">
                                            <div className="w-1.5 h-1.5 rounded-full mb-1" style={{ backgroundColor: s.color }} />{s.name}
                                        </div>
                                    ))}
                                    {Array.from({ length: (dayRange.end - dayRange.start) / TIME_STEP }).map((_, i) => (
                                        <div key={i} className="sticky left-0 bg-slate-50 border-r border-b border-slate-100 text-[8px] font-black flex items-center justify-center text-slate-400 z-10 font-mono">{minutesToTime(dayRange.start + (i * TIME_STEP))}</div>
                                    ))}
                                    {acts.filter(a => a.day === currentDay).map(act => {
                                        const stage = stages.find(s => s.id === act.stageId);
                                        const isPicked = selectedIds.includes(act.id);
                                        return (
                                            <div key={act.id} onClick={() => togglePick(act.id)} className={`m-[1px] p-1 text-[8px] leading-[1.1] font-black rounded-md flex items-center justify-center text-center transition-all cursor-pointer border-2 ${isPicked ? 'z-40 shadow-lg' : 'opacity-20 grayscale'}`} style={{ gridRowStart: (timeToMinutes(act.startTime)-dayRange.start)/TIME_STEP+2, gridRowEnd: (timeToMinutes(act.endTime)-dayRange.start)/TIME_STEP+2, gridColumnStart: stages.findIndex(s=>s.id===act.stageId)+2, backgroundColor: isPicked ? `${stage?.color}20` : '#f1f5f9', borderColor: isPicked ? stage?.color : '#e2e8f0', color: isPicked ? stage?.color : '#94a3b8' }}>
                                                {act.name}
                                            </div>
                                        );
                                    })}
                                </div>
                            </div>
                        )}

                        {activeTab === 'picks' && (
                            <div className="space-y-4">
                                {acts.filter(a => selectedIds.includes(a.id) && a.day === currentDay).sort((a,b) => timeToMinutes(a.startTime) - timeToMinutes(b.startTime)).map((act, idx, arr) => {
                                    const conflict = arr[idx-1] && timeToMinutes(act.startTime) < timeToMinutes(arr[idx-1].endTime);
                                    return (
                                        <div key={act.id} className={`p-6 rounded-[32px] bg-white shadow-sm border-2 ${conflict ? 'border-red-500 bg-red-50' : 'border-white'}`}>
                                            <div className="flex justify-between items-start">
                                                <div className="space-y-2">
                                                    <div className="flex items-center gap-2">
                                                        <span className="text-xs font-black font-mono bg-slate-900 text-white px-3 py-1 rounded-full">{act.startTime} - {act.endTime}</span>
                                                        {conflict && <span className="text-[10px] bg-red-500 text-white px-2 py-0.5 rounded-full font-black animate-pulse">CONFLICT</span>}
                                                    </div>
                                                    <h3 className="text-xl font-black">{act.name}</h3>
                                                    <div className="flex items-center gap-2 text-[10px] font-black uppercase tracking-widest" style={{ color: stages.find(s=>s.id===act.stageId)?.color }}>
                                                        <MapPin size={12} /> {stages.find(s=>s.id===act.stageId)?.name}
                                                    </div>
                                                </div>
                                            </div>
                                        </div>
                                    );
                                })}
                            </div>
                        )}
                    </main>

                    {isAdminOpen && (
                        <div className="fixed inset-0 bg-slate-900/40 backdrop-blur-md z-[100] flex items-end">
                            <div className="bg-white w-full rounded-t-[40px] p-8 max-h-[90vh] overflow-y-auto">
                                <div className="flex justify-between items-center mb-8">
                                    <h2 className="text-2xl font-black italic">CONFIG.</h2>
                                    <button onClick={() => setIsAdminOpen(false)} className="p-2 bg-slate-100 rounded-full"><X size={24}/></button>
                                </div>
                                <section className="mb-10">
                                    <h3 className="text-[10px] font-black text-slate-400 uppercase mb-4 tracking-[0.2em]">1. 舞台設定</h3>
                                    <div className="bg-slate-50 p-6 rounded-[24px] space-y-4">
                                        <input id="s-name" type="text" placeholder="舞台名稱..." className="w-full bg-white border-none rounded-xl px-4 py-3 text-sm font-bold" />
                                        <div className="flex flex-wrap gap-2">
                                            {PRESET_COLORS.map(c => (
                                                <button key={c} onClick={() => window._selColor = c} className="w-8 h-8 rounded-full border-2 border-white shadow-sm focus:ring-2 focus:ring-black" style={{ backgroundColor: c }} />
                                            ))}
                                        </div>
                                        <button onClick={() => { const n = document.getElementById('s-name').value; if(n) setStages([...stages, { id: Date.now().toString(), name: n, color: window._selColor || '#000' }]); document.getElementById('s-name').value=''; }} className="w-full bg-black text-white py-3 rounded-xl font-black text-xs uppercase">建立舞台</button>
                                    </div>
                                    <div className="mt-4 flex flex-wrap gap-2">
                                        {stages.map(s => (
                                            <div key={s.id} className="bg-white border-2 px-3 py-1.5 rounded-full text-[10px] font-black flex items-center gap-2 shadow-sm" style={{ borderColor: s.color }}>
                                                <div className="w-2 h-2 rounded-full" style={{ backgroundColor: s.color }} />{s.name}
                                                <Trash2 size={12} className="text-slate-300" onClick={() => setStages(stages.filter(x=>x.id!==s.id))}/>
                                            </div>
                                        ))}
                                    </div>
                                </section>
                                <section>
                                    <h3 className="text-[10px] font-black text-slate-400 uppercase mb-4 tracking-[0.2em]">2. 演出資料</h3>
                                    <form className="space-y-4" onSubmit={(e) => {
                                        e.preventDefault();
                                        const f = e.target;
                                        setActs([...acts, { id: Date.now().toString(), day: f.day.value, stageId: f.stageId.value, name: f.actName.value, startTime: `${f.sh.value}:${f.sm.value}`, endTime: `${f.eh.value}:${f.em.value}` }]);
                                        f.actName.value = '';
                                    }}>
                                        <div className="grid grid-cols-2 gap-3">
                                            <input name="day" type="date" required className="bg-slate-50 border-none rounded-xl px-4 py-3 text-xs font-bold" />
                                            <select name="stageId" className="bg-slate-50 border-none rounded-xl px-4 py-3 text-xs font-bold">{stages.map(s => <option key={s.id} value={s.id}>{s.name}</option>)}</select>
                                        </div>
                                        <input name="actName" placeholder="樂團名稱" required className="w-full bg-slate-50 border-none rounded-xl px-4 py-3 text-xs font-bold" />
                                        <div className="grid grid-cols-2 gap-4 bg-slate-50 p-4 rounded-2xl text-center">
                                            <div><p className="text-[9px] font-black text-slate-400 mb-2">START</p><div className="flex gap-1"><TimeSelect name="sh" type="h" /><TimeSelect name="sm" type="m" /></div></div>
                                            <div><p className="text-[9px] font-black text-slate-400 mb-2">END</p><div className="flex gap-1"><TimeSelect name="eh" type="h" /><TimeSelect name="em" type="m" /></div></div>
                                        </div>
                                        <button type="submit" className="w-full bg-orange-500 text-white font-black py-4 rounded-2xl shadow-xl uppercase tracking-widest text-xs flex items-center justify-center gap-2"><Save size={18}/> 存入時程</button>
                                    </form>
                                </section>
                            </div>
                        </div>
                    )}
                </div>
            );
        }

        function TimeSelect({ name, type }) {
            const opts = type === 'h' ? Array.from({length: 24}, (_, i) => i.toString().padStart(2, '0')) : ['00', '10', '20', '30', '40', '50'];
            return <select name={name} className="flex-1 bg-white rounded-lg p-2 text-xs font-black">{opts.map(o => <option key={o} value={o}>{o}</option>)}</select>;
        }

        function TabBtn({ active, onClick, icon, label }) {
            return (
                <button onClick={onClick} className={`flex-1 py-4 flex flex-col items-center gap-1 transition-all relative ${active ? 'text-black' : 'text-slate-300'}`}>
                    <i data-lucide={icon} className="w-5 h-5"></i>
                    <span className="text-[10px] font-black uppercase tracking-[0.1em]">{label}</span>
                    {active && <div className="absolute bottom-0 w-10 h-1 bg-black rounded-full" />}
                </button>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
</html>
